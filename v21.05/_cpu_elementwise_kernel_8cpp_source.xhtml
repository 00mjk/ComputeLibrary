160 Status CpuElementwiseKernel::validate_arguments_common(const ITensorInfo &src0, const ITensorInfo &src1, const ITensorInfo &dst)
161 {
162  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(&src0);
163  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(&src0, &src1);
164 
165  const TensorShape out_shape = TensorShape::broadcast_shape(src0.tensor_shape(), src1.tensor_shape());
166 
167  ARM_COMPUTE_RETURN_ERROR_ON_MSG(out_shape.total_size() == 0, "Inputs are not broadcast compatible");
168 
169  // Validate in case of configured dst
170  if(dst.total_size() > 0)
171  {
172  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(out_shape, dst.tensor_shape(), 0),
173  "Wrong shape for output");
174  }
175 
176  return Status{};
177 }

179 void CpuElementwiseKernel::configure_common(const ITensorInfo *src0, const ITensorInfo *src1, ITensorInfo *dst)
180 {
181  ARM_COMPUTE_ERROR_ON_NULLPTR(src0, src1, dst);
182 
183  // If any of shapes is dynamic, expect a configured window and dst at run-time.
184  if(src0->is_dynamic() || src1->is_dynamic())
185  {
186  return;
187  }
188 
189  auto shape_and_window = compute_output_shape_and_window(src0->tensor_shape(), src1->tensor_shape());
190  auto_init_if_empty(*dst, shape_and_window.first, 1, src0->data_type());
191  ICpuKernel::configure(shape_and_window.second);
192 }

194 void CpuElementwiseKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
195 {
196  ARM_COMPUTE_UNUSED(info);
197 
198  auto src0 = tensors.get_const_tensor(TensorType::ACL_SRC_0);
199  auto src1 = tensors.get_const_tensor(TensorType::ACL_SRC_1);
200  auto dst = tensors.get_tensor(TensorType::ACL_DST);
201 
202  auto function = get_implementation(src0->info(), src1->info(), dst->info());
203  ARM_COMPUTE_ERROR_ON(function == nullptr);
204  function(src0, src1, dst, window);
205 }

234 std::function<CpuElementwiseKernel::ElementwiseFunction>
235 CpuArithmeticKernel::get_implementation(const ITensorInfo *src0, const ITensorInfo *src1, ITensorInfo *dst)
236 {
