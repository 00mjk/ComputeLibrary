42 CLGEMMKernelType CLGEMMKernelSelectionBifrost::select_kernel(const CLGEMMKernelSelectionParams &params)
43 {
44  // _target could be used in the future to have a dedicated heuristic for each GPU IP
45  ARM_COMPUTE_UNUSED(_target);
46 
47  using FunctionExecutorPtr = CLGEMMKernelType (CLGEMMKernelSelectionBifrost::*)(unsigned int m, unsigned int n, unsigned int k, bool is_rhs_constant);
48 
49  // Default configurations for Bifrost architectures
50  static std::map<DataType, FunctionExecutorPtr> gemm_default_configs =
51  {
52  { DataType::F32, &CLGEMMKernelSelectionBifrost::default_f32 },
53  { DataType::F16, &CLGEMMKernelSelectionBifrost::default_f16 },
54  { DataType::QASYMM8, &CLGEMMKernelSelectionBifrost::default_q8 },
55  { DataType::QASYMM8_SIGNED, &CLGEMMKernelSelectionBifrost::default_q8 },
56  { DataType::QSYMM8, &CLGEMMKernelSelectionBifrost::default_q8 },
57  { DataType::QSYMM8_PER_CHANNEL, &CLGEMMKernelSelectionBifrost::default_q8 }
58  };
59 
60  // Mali-G71 configurations
61  static std::map<DataType, FunctionExecutorPtr> gemm_g71_configs =
62  {
63  { DataType::F32, &CLGEMMKernelSelectionBifrost::default_f32 },
64  { DataType::F16, &CLGEMMKernelSelectionBifrost::g71_f16 },
65  { DataType::QASYMM8, &CLGEMMKernelSelectionBifrost::default_q8 },
66  { DataType::QASYMM8_SIGNED, &CLGEMMKernelSelectionBifrost::default_q8 },
67  { DataType::QSYMM8, &CLGEMMKernelSelectionBifrost::default_q8 },
68  { DataType::QSYMM8_PER_CHANNEL, &CLGEMMKernelSelectionBifrost::default_q8 }
69  };
70 
71  // Mali-G76 configurations
72  static std::map<DataType, FunctionExecutorPtr> gemm_g76_configs =
73  {
74  { DataType::F32, &CLGEMMKernelSelectionBifrost::g76_f32 },
75  { DataType::F16, &CLGEMMKernelSelectionBifrost::default_f16 },
76  { DataType::QASYMM8, &CLGEMMKernelSelectionBifrost::default_q8 },
77  { DataType::QASYMM8_SIGNED, &CLGEMMKernelSelectionBifrost::default_q8 },
78  { DataType::QSYMM8, &CLGEMMKernelSelectionBifrost::default_q8 },
79  { DataType::QSYMM8_PER_CHANNEL, &CLGEMMKernelSelectionBifrost::default_q8 }
80  };
81 
82  const DataType data_type = params.data_type;
83 
84  switch(_target)
85  {
86  case GPUTarget::G71:
87  if(gemm_g71_configs.find(data_type) != gemm_g71_configs.end())
88  {
89  return (this->*gemm_g71_configs[data_type])(params.m, params.n, params.k, params.is_rhs_constant);
90  }
91  ARM_COMPUTE_ERROR("Not supported data type");
92  case GPUTarget::G76:
93  if(gemm_g76_configs.find(data_type) != gemm_g76_configs.end())
94  {
95  return (this->*gemm_g76_configs[data_type])(params.m, params.n, params.k, params.is_rhs_constant);
96  }
97  ARM_COMPUTE_ERROR("Not supported data type");
98  default:
99  if(gemm_default_configs.find(data_type) != gemm_default_configs.end())
100  {
101  return (this->*gemm_default_configs[data_type])(params.m, params.n, params.k, params.is_rhs_constant);
102  }
103  ARM_COMPUTE_ERROR("Not supported data type");
104  }
105 }

107 CLGEMMKernelType CLGEMMKernelSelectionBifrost::default_f32(unsigned int m, unsigned int n, unsigned int k, bool is_rhs_constant)
108 {
109  CLGEMMKernelType gemm_type = CLGEMMKernelType::NATIVE_V1;
110 
111  if(is_rhs_constant)
112  {
113  if((m > 1) && (n < 16))
114  {
115  gemm_type = CLGEMMKernelType::RESHAPED_V1;
116  }
117  else if(m == 1)
118  {
119  gemm_type = CLGEMMKernelType::RESHAPED_ONLY_RHS;
120  }
121  else
122  {
123  if((k > 256) && (m > 4))
124  {
125  constexpr float alpha = 3.2f;
126  constexpr float fact0 = 1.51f;
127  constexpr float fact1 = 1.66f;
128  constexpr float ops = 12.0f;
129  const float scale = k > 1024 ? 1.07f : 1.0f;
130  gemm_type = (alpha + ((n * fact0) / ops) < ((fact1 * n * scale) / ops)) ? CLGEMMKernelType::RESHAPED_V1 : CLGEMMKernelType::NATIVE_V1;
131  }
132  else
133  {
134  gemm_type = CLGEMMKernelType::NATIVE_V1;
135  }
136  }
137 
138  const auto workload = static_cast<float>((m * n) / 20.0f);
139 
140  gemm_type = ((workload > 1600.0f) && (gemm_type == CLGEMMKernelType::RESHAPED_V1)) ? CLGEMMKernelType::RESHAPED : gemm_type;
141  }
142 
143  return gemm_type;
144 }
