37 void configure_border_handler(const CLCompileContext &compile_context, CLFillBorderKernel &border_handler, BorderSize border_size, ITensorInfo *input1, ITensorInfo *input2, const ITensorInfo *output)
38 {
39  if(output->dimension(0) > 1)
40  {
41  ITensorInfo *broadcasted_info = (input1->dimension(0) == 1) ? input1 : input2;
42 
43  if(broadcasted_info->dimension(0) == 1)
44  {
45  border_handler.configure(compile_context, broadcasted_info, border_size, BorderMode::REPLICATE);
46  }
47  }
48 }
49 
50 ITensorPack select_border_input(ITensorPack &tensors)
51 {
52  ITensorPack pack;
53  if(tensors.get_tensor(TensorType::ACL_DST)->info()->dimension(0) > 1)
54  {
55  if(tensors.get_const_tensor(TensorType::ACL_SRC_1)->info()->dimension(0) == 1)
56  {
57  pack.add_tensor(TensorType::ACL_SRC, tensors.get_const_tensor(TensorType::ACL_SRC_1));
58  }
59  else
60  {
61  pack.add_tensor(TensorType::ACL_SRC, tensors.get_const_tensor(TensorType::ACL_SRC_0));
62  }
63  }
64  return pack;
65 }
