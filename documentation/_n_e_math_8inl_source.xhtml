75 {

86 inline float32x4_t vexpq_f32(float32x4_t x)
87 {
88  static const float32x4_t CONST_LN2 = vdupq_n_f32(0.6931471805f); // ln(2)
89  static const float32x4_t CONST_INV_LN2 = vdupq_n_f32(1.4426950408f); // 1/ln(2)
90 
91  // Perform range reduction [-log(2),log(2)]
92  int32x4_t m = vcvtq_s32_f32(vmulq_f32(x, CONST_INV_LN2));
93  float32x4_t val = vmlsq_f32(x, vcvtq_f32_s32(m), CONST_LN2);
94 
95  // Polynomial Approximation
96  float32x4_t poly = vtaylor_polyq_f32(val, exp_tab);
97 
98  // Reconstruct
99  poly = vreinterpretq_f32_s32(vaddq_s32(vreinterpretq_s32_f32(poly), vshlq_n_s32(m, 23)));
100 
101  return poly;
102 }

105 {

108 
109  // Extract exponent
110  int32x4_t m = vsubq_s32(vreinterpretq_s32_u32(vshrq_n_u32(vreinterpretq_u32_f32(x), 23)), CONST_127);
111  float32x4_t val = vreinterpretq_f32_s32(vsubq_s32(vreinterpretq_s32_f32(x), vshlq_n_s32(m, 23)));

122 inline float32x4_t vtanhq_f32(float32x4_t val)
123 {
124  static const float32x4_t CONST_1 = vdupq_n_f32(1.f);
125  static const float32x4_t CONST_2 = vdupq_n_f32(2.f);
126  static const float32x4_t CONST_MIN_TANH = vdupq_n_f32(-10.f);
127  static const float32x4_t CONST_MAX_TANH = vdupq_n_f32(10.f);
128 
129  float32x4_t x = vminq_f32(vmaxq_f32(val, CONST_MIN_TANH), CONST_MAX_TANH);
130  float32x4_t exp2x = vexpq_f32(vmulq_f32(CONST_2, x));
131  float32x4_t num = vsubq_f32(exp2x, CONST_1);
132  float32x4_t den = vaddq_f32(exp2x, CONST_1);
133  float32x4_t tanh = vmulq_f32(num, vinvq_f32(den));
134  return tanh;
135 }
