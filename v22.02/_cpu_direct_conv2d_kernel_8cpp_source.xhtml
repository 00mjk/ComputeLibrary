55 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *weights, const ITensorInfo *dst, const PadStrideInfo &conv_info)
56 {
57  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, weights, dst);
58  ARM_COMPUTE_RETURN_ERROR_ON(src->data_layout() == DataLayout::UNKNOWN);
59  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(src);
60  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::F16, DataType::F32);
61  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, weights);
62 
63  const DataLayout data_layout = src->data_layout();
64  const int width_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
65  const int height_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
66  const int channel_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
67 
68  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(channel_idx) != src->dimension(channel_idx));
69  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(width_idx) != weights->dimension(height_idx));
70  ARM_COMPUTE_RETURN_ERROR_ON(weights->num_dimensions() > 4);
71  ARM_COMPUTE_RETURN_ERROR_ON(data_layout == DataLayout::NHWC && src->data_type() != DataType::F32);
72  ARM_COMPUTE_UNUSED(width_idx);
73  // Checks performed when output is configured
74  if(dst->total_size() != 0)
75  {
76  TensorShape output_shape = misc::shape_calculator::compute_deep_convolution_shape(*src, *weights, conv_info);
77 
78  DataType data_type = src->data_type();
79 
80  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(dst->tensor_shape(), output_shape);
81  ARM_COMPUTE_RETURN_ERROR_ON(dst->data_type() != data_type);
82  }
83 
84  return Status{};
85 }

109 template <typename T>
110 void CpuDirectConv2dKernel::convolve_nhwc_optimized(const Window &window, const ITensor *src, const ITensor *weights, ITensor *dst)
111 {
112  // This function assumes that input and weights have not padding in channel
113 
114  // Declare useful types
115  using vtype = wrapper::traits::neon_bitvector<T, wrapper::traits::BitWidth::W128>;
116  using vector_type = typename vtype::type;
117  using tag_type = typename vtype::tag_type;
118 
119  // Scalar quantities
120  const int element_size = src->info()->element_size();
121  const int input_stride_w = src->info()->strides_in_bytes().y() / element_size;
122  const int input_stride_h = src->info()->strides_in_bytes().z() / element_size;
123  const int input_stride_n = src->info()->strides_in_bytes()[3] / element_size;
124  const int input_dim_w = src->info()->dimension(1);
125  const int input_dim_h = src->info()->dimension(2);
126 
127  const int output_stride_c = dst->info()->strides_in_bytes().x();
128 
129  const unsigned int kernel_stride_w = weights->info()->strides_in_bytes().y() / element_size;
130  const unsigned int kernel_stride_h = weights->info()->strides_in_bytes().z() / element_size;
131  const int kernel_dim_w = weights->info()->dimension(1);
132  const int kernel_dim_h = weights->info()->dimension(2);
133 
134  const int conv_pad_top = _conv_info.pad_top();
135  const int conv_pad_left = _conv_info.pad_left();
136  const int conv_stride_w = std::get<0>(_conv_info.stride());
137  const int conv_stride_h = std::get<1>(_conv_info.stride());
138 
139  // Setup input window for the output iterator
140  Window window_out = window;
141  window_out.set(Window::DimX, Window::Dimension(0, 1, 1));
142 
143  // Setup input window for the weights iterator
144  Window window_w = calculate_max_window(*weights->info(), Steps());
145  window_w.set(Window::DimX, Window::Dimension(0, 1, 1));
146  window_w.set(Window::DimY, Window::Dimension(0, 1, 1));
147  window_w.set(Window::DimZ, Window::Dimension(0, 1, 1));
148 
149  Iterator out(dst, window_out);
150  Iterator wei(weights, window_w);
151 
152  constexpr int num_elems_read_per_iteration = 16 / sizeof(T);
153  /*
154  * This implementation parallelize the full WC plane of input and weights by
155  * treating them as series of elements. So for example, a 3x3 weights and
156  * floating point vector operations of 4 elements per time, the first 3
157  * channel elements of the first row would be taken and additionally the first
158  * element of the second row. The 9 elements in each single WC weight plane
159  * would require 2 4-element vector operations and a last single element operation.
160  *
161  * This works since when we create the input vector to multiply with the weights,
162  * the exact required elements are loaded in the same order. Therefore the
163  * multiplication works on the correct input/weight elements.
164  */
165  execute_window_loop(window_out, [&](const Coordinates & id)
166  {
167  /*
168  * In here we create theoretical indexes which then we validate for both
169  * inputs and weights.
170  * As a reminder, this loop take each output point in NHW, C is treated
171  * in the weights loop.
172  */
173  // We are computing the theoretical starting input starting points
174  const int in_w_start_t = static_cast<int>(id.y()) * conv_stride_w - conv_pad_left;
175  const int in_h_start_t = static_cast<int>(id.z()) * conv_stride_h - conv_pad_top;
176  const int in_w_end_t = in_w_start_t + kernel_dim_w;
177  const int in_h_end_t = in_h_start_t + kernel_dim_h;
178 
179  // We are computing the valid initial and ending input points by checking the borders
180  const int in_w_start = std::max(in_w_start_t, 0);
181  const int in_h_start = std::max(in_h_start_t, 0);
182  const int in_w_end = std::min(in_w_end_t, input_dim_w);
183  const int in_h_end = std::min(in_h_end_t, input_dim_h);
184 
185  // We use the input points to select the valid weight points to use
186  const int index_wc_start = (in_w_start - in_w_start_t) * kernel_stride_w;
187  const int index_h_start = in_h_start - in_h_start_t;
188  const int index_wc_end = (kernel_dim_w - (in_w_end_t - in_w_end)) * kernel_stride_w;
189  const int index_h_end = kernel_dim_h - (in_h_end_t - in_h_end);
190 
191  execute_window_loop(window_w, [&](const Coordinates & id_w)
192  {
193  /*
194  * This is the loop in the weights, and it goes along N (the batches)
195  * As a reminder, the batches of the weights are translated into the
196  * channels of the output
197  */
198  const T *in_ptr_row = reinterpret_cast<const T *>(src->buffer() + src->info()->offset_first_element_in_bytes())
199  + id[3] * input_stride_n + in_w_start * input_stride_w + in_h_start * input_stride_h;
200  const T *weights_ptr_row = reinterpret_cast<const T *>(wei.ptr()) + index_h_start * kernel_stride_h;
201  uint8_t *out_ptr = out.ptr() + id_w[3] * output_stride_c;
202 
203  T out_temp = static_cast<T>(0);
204  for(int index_h = index_h_start; index_h < index_h_end; ++index_h, in_ptr_row += input_stride_h, weights_ptr_row += kernel_stride_h)
205  {
206  const T *in_ptr_mover = in_ptr_row;
207  int index_wc = index_wc_start;
208  vector_type out_temp_vec = wrapper::vdup_n(static_cast<T>(0), tag_type());
209  for(; index_wc <= index_wc_end - num_elems_read_per_iteration; index_wc += num_elems_read_per_iteration, in_ptr_mover += num_elems_read_per_iteration)
210  {
211  const auto src_vec = wrapper::vloadq(in_ptr_mover);
212  const auto w_vec = wrapper::vloadq(weights_ptr_row + index_wc);
213  out_temp_vec = wrapper::vmla(out_temp_vec, w_vec, src_vec);
214  }
215  out_temp += vreduce(out_temp_vec);
216  for(; index_wc < index_wc_end; ++index_wc, ++in_ptr_mover)
217  {
218  const auto src_val = *(in_ptr_mover);
219  const auto w_val = *(weights_ptr_row + index_wc);
220  out_temp += src_val * w_val;
221  }
222  }
223  *(reinterpret_cast<T *>(out_ptr)) = out_temp;
224  },
225  wei);
226  },
227  out);
228 }

251  const int kernel_dim_h = weights->info()->dimension(2);
252

254  const int conv_pad_left = _conv_info.pad_left();
255  const int conv_stride_w = std::get<0>(_conv_info.stride());
256  const int conv_stride_h = std::get<1>(_conv_info.stride());
257 
258  // Setup input window for the output iterator
259  Window window_out = window;
260  window_out.set(Window::DimX, Window::Dimension(0, 1, 1));
261 
262  // Setup input window for the weights iterator
263  Window window_w = calculate_max_window(*weights->info(), Steps());
264  window_w.set(Window::DimX, Window::Dimension(0, 1, 1));
265  window_w.set(Window::DimY, Window::Dimension(0, 1, 1));
266  window_w.set(Window::DimZ, Window::Dimension(0, 1, 1));
267 
268  Iterator out(dst, window_out);

270 
271  constexpr int num_elems_read_per_iteration = 16 / sizeof(T);
272 
273  execute_window_loop(window_out, [&](const Coordinates & id)
274  {
275  // We are computing the theoretical starting input starting points
276  const int in_w_start_t = static_cast<int>(id.y()) * conv_stride_w - conv_pad_left;
277  const int in_h_start_t = static_cast<int>(id.z()) * conv_stride_h - conv_pad_top;
278  const int in_w_end_t = in_w_start_t + kernel_dim_w;
279  const int in_h_end_t = in_h_start_t + kernel_dim_h;
280 
281  // We are computing the valid initial and ending input points by checking the borders
282  const int in_w_start = std::max(in_w_start_t, 0);
283  const int in_h_start = std::max(in_h_start_t, 0);
284  const int in_w_end = std::min(in_w_end_t, input_dim_w);
285  const int in_h_end = std::min(in_h_end_t, input_dim_h);
286 
287  // We use the input points to select the valid weight points to use
288  const int wei_w_start = in_w_start - in_w_start_t;
289  const int wei_h_start = in_h_start - in_h_start_t;
290  const int wei_w_end = kernel_dim_w - (in_w_end_t - in_w_end);
291  const int wei_h_end = kernel_dim_h - (in_h_end_t - in_h_end);
292 
293  const int index_c_end = weights->info()->dimension(0);
294  const T *const in_ptr_start = reinterpret_cast<const T *>(src->buffer() + src->info()->offset_first_element_in_bytes()) + id[3] * input_stride_n;
295 
296  execute_window_loop(window_w, [&](const Coordinates & id_w)
297  {
298  const T *const weights_ptr_start = reinterpret_cast<const T *>(wei.ptr());
299  uint8_t *out_ptr = out.ptr() + id_w[3] * output_stride_c;
300 
301  T out_temp = static_cast<T>(0);
302  for(int index_wei_h = wei_h_start, index_in_h = in_h_start; index_wei_h < wei_h_end; ++index_wei_h, ++index_in_h)
303  {
304  const T *const in_ptr_row = in_ptr_start + index_in_h * input_stride_h;
305  const T *const weights_ptr_row = weights_ptr_start + index_wei_h * kernel_stride_h;
306  for(int index_wei_w = wei_w_start, index_in_w = in_w_start; index_wei_w < wei_w_end; ++index_wei_w, ++index_in_w)
307  {
308  const T *in_ptr_mover = in_ptr_row + index_in_w * input_stride_w;
309  const T *weights_ptr_mover = weights_ptr_row + index_wei_w * kernel_stride_w;
310  int index_c = 0;
311  vector_type out_temp_vec = wrapper::vdup_n(static_cast<T>(0), tag_type());
312  for(; index_c <= index_c_end - num_elems_read_per_iteration; index_c += num_elems_read_per_iteration, in_ptr_mover += num_elems_read_per_iteration, weights_ptr_mover += num_elems_read_per_iteration)
313  {
314  const auto src_vec = wrapper::vloadq(in_ptr_mover);
315  const auto w_vec = wrapper::vloadq(weights_ptr_mover);
316  out_temp_vec = wrapper::vmla(out_temp_vec, w_vec, src_vec);
317  }
318  out_temp += vreduce(out_temp_vec);
319  for(; index_c < index_c_end; ++index_c, ++in_ptr_mover, ++weights_ptr_mover)
320  {
321  const auto src_val = *(in_ptr_mover);
322  const auto w_val = *(weights_ptr_mover);
323  out_temp += src_val * w_val;
324  }
325  }
326  }
327  *(reinterpret_cast<T *>(out_ptr)) = out_temp;
328  },
329  wei);
330  },
331  out);
332 }

442 void CpuDirectConv2dKernel::configure(ITensorInfo *src, ITensorInfo *weights, ITensorInfo *dst, const PadStrideInfo &conv_info)
443 {
444  ARM_COMPUTE_ERROR_ON_NULLPTR(src, weights, dst);
445 
446  _conv_info = conv_info;
447  _data_layout = src->data_layout();
448  _kernel_size = weights->dimension(get_data_layout_dimension_index(_data_layout, DataLayoutDimension::WIDTH));
449 
450  // Get convolved dimensions
451  TensorShape output_shape = misc::shape_calculator::compute_deep_convolution_shape(*src, *weights, conv_info);
