51 ACC_DATA_TYPE calculate_avg_scale(const int pool_size_x, const int pool_size_y, const int upper_bound_w, const int upper_bound_h,
52  const int pad_x, const int pad_y, const int stride_x, const int stride_y)
53 {
54  int start_x = get_global_id(0) * stride_x - pad_x;
55  int start_y = get_global_id(1) * stride_y - pad_y;
56  const int end_x = min(start_x + pool_size_x, upper_bound_w);
57  const int end_y = min(start_y + pool_size_y, upper_bound_h);
58 #if defined(EXCLUDE_PADDING)
59  start_x = max(0, start_x);
60  start_y = max(0, start_y);
61 #endif /* defined(EXCLUDE_PADDING) */
62  return ((end_y - start_y) * (end_x - start_x));
63 }

67 /** Performs a pooling function of pool size equal to N (NCHW)
68  *
69  * @note Datatype must be passed using -DDATA_TYPE e.g. -DDATA_TYPE=float. Supported data types are F16/F32;
70  * @note Pool sizes must be passed using -DPOOL_SIZE_X and -DPOOL_SIZE_Y e.g. -DPOOL_SIZE_X=13;
71  * @note In case of average pooling the following information must be passed at compile time:
72  * -DPOOL_AVG must be provided otherwise max pooling will be performed.
73  * -DMAX_WIDTH and -DMAX_HEIGHT which are the maximum accessible indeces in x and y dimensions (width + pad)
74  * -DSTRIDE_X and -DSTRIDE_Y which are the steps of the window along the x and y directions
75  * -DPAD_X and -DPAD_Y which are the pooling paddings in x and y dimension
76  * @note The initial value for the pooling operation must be passed at compile time using -DINITIAL_VALUE e.g. -DINITIAL_VALUE=0
77  *
78  * @param[in] input_ptr Pointer to the source tensor. Supported data types: F16/F32
79  * @param[in] input_stride_x Stride of the source tensor in X dimension (in bytes)
80  * @param[in] input_step_x input_stride_x * number of elements along X processed per workitem(in bytes)
81  * @param[in] input_stride_y Stride of the source tensor in Y dimension (in bytes)
82  * @param[in] input_step_y input_stride_y * number of elements along Y processed per workitem(in bytes)
83  * @param[in] input_stride_z Stride of the source tensor in Z dimension (in bytes)
84  * @param[in] input_step_z input_stride_z * number of elements along Z processed per workitem(in bytes)
85  * @param[in] input_offset_first_element_in_bytes The offset of the first element in the source tensor
86  * @param[out] output_ptr Pointer to the destination tensor. Supported data types: same as @p input_ptr
87  * @param[in] output_stride_x Stride of the destination tensor in X dimension (in bytes)
88  * @param[in] output_step_x output_stride_x * number of elements along X processed per workitem(in bytes)
89  * @param[in] output_stride_y Stride of the destination tensor in Y dimension (in bytes)
90  * @param[in] output_step_y output_stride_y * number of elements along Y processed per workitem(in bytes)
91  * @param[in] output_stride_z Stride of the source tensor in Z dimension (in bytes)
92  * @param[in] output_step_z output_stride_z * number of elements along Z processed per workitem(in bytes)
93  * @param[in] output_offset_first_element_in_bytes The offset of the first element in the destination tensor
94  */
95 __kernel void pooling_layer_MxN_nchw(
96  TENSOR3D_DECLARATION(input),
97  TENSOR3D_DECLARATION(output))
98 {
99  // Get pixels pointer
100  Tensor3D input = CONVERT_TO_TENSOR3D_STRUCT(input);
101  Tensor3D output = CONVERT_TO_TENSOR3D_STRUCT(output);
102 
103  VEC_DATA_TYPE(ACC_DATA_TYPE, 8)
104  vdata = INITIAL_VALUE;
105  ACC_DATA_TYPE sdata = INITIAL_VALUE;
106 
107  // Load data
108  for(int y = 0; y < POOL_SIZE_Y; y++)
109  {
110  int x = 0;
111  for(; x <= ((int)POOL_SIZE_X - 8); x += 8)
112  {
113  VEC_DATA_TYPE(ACC_DATA_TYPE, 8)
114  data0 = VLOAD_AND_CONVERT_TO_ACC_DATA_TYPE(8, 0, (__global DATA_TYPE *)tensor3D_offset(&input, x, y, 0));
115 #if defined(POOL_L2)
116  // Raise to power of 2 for L2 Pooling
117  data0 *= data0;
118 #endif /* defined(POOL_L2) */
119  vdata = POOL_OP(vdata, data0);
120  }
121 
122  // Leftover
123  for(; x < (int)POOL_SIZE_X; ++x)
124  {
125  ACC_DATA_TYPE data0 = (ACC_DATA_TYPE)(*((__global DATA_TYPE *)tensor3D_offset(&input, x, y, 0)));
126 #if defined(POOL_L2)
127  // Raise to power of 2 for L2 Pooling
128  data0 *= data0;
129 #endif /* defined(POOL_L2) */
130  sdata = POOL_OP(sdata, data0);
131  }
132  }
133 
134  // Reduce result
135  VEC_DATA_TYPE(ACC_DATA_TYPE, 4)
136  reduce4 = POOL_OP(vdata.s0123, vdata.s4567);
137  VEC_DATA_TYPE(ACC_DATA_TYPE, 2)
138  reduce2 = POOL_OP(reduce4.s01, reduce4.s23);
139  ACC_DATA_TYPE res = POOL_OP(reduce2.s0, reduce2.s1);
140  res = POOL_OP(res, sdata);
141 
142 #if defined(POOL_AVG) || defined(POOL_L2)
143  // Divide by pool region in case of average pooling
144  res = DIV_OP(res, calculate_avg_scale(POOL_SIZE_X, POOL_SIZE_Y, MAX_WIDTH, MAX_HEIGHT, PAD_X, PAD_Y, STRIDE_X, STRIDE_Y));
145 #endif /* defined(POOL_AVG) || defined(POOL_L2) */
146 
147 #if defined(POOL_L2)
148  // Take square root of the result in L2 pooling
149  res = SQRT_OP(res);
150 #endif /* defined(POOL_L2) */
151 
152  // Store result
153  *(__global DATA_TYPE *)output.ptr = (DATA_TYPE)res;
154 }
155 #endif // defined(POOL_SIZE_X) && defined(POOL_SIZE_Y)

159 inline void offset_no_padding_nchw(const Tensor3D *input, uint *offset_top, uint *offset_bottom)
160 {
161  const int pad_horiz = PAD_TENSOR_LEFT + PAD_TENSOR_RIGHT;
162  const int pad_vert = PAD_TENSOR_TOP + PAD_TENSOR_BOTTOM;
163 
164  const int x = get_global_id(0) * STRIDE_X;
165  const int y = get_global_id(1) * STRIDE_Y;
166  const int z = get_global_id(2);
167 
168  //x axis: width, y axis: height, z axis: component
169  const uint padded_offset = input->offset_first_element_in_bytes
170  + x * input->stride_x
171  + y * input->stride_y
172  + z * input->stride_z;
173 
174  const uint offset_base = padded_offset
175  - y * pad_horiz * sizeof(DATA_TYPE) /* Horizontal padding for each row */
176  - PAD_TENSOR_TOP * input->stride_y /* top padding */
177  - z * MAX_HEIGHT * pad_horiz * sizeof(DATA_TYPE) - z * pad_vert * input->stride_y /* Z plane padding */
178  - PAD_TENSOR_LEFT * sizeof(DATA_TYPE);
179 
180 #if defined(TENSOR_CHANNEL) && defined(TENSOR_WIDTH) && defined(TENSOR_HEIGHT)
181  *offset_top = (uint)((offset_base / sizeof(DATA_TYPE)) % (TENSOR_CHANNEL * TENSOR_WIDTH * TENSOR_HEIGHT));
182 #else /* defined(TENSOR_CHANNEL) && defined(TENSOR_WIDTH) && defined(TENSOR_HEIGHT) */
183  *offset_top = (uint)(offset_base / sizeof(DATA_TYPE));
184 #endif /* defined(TENSOR_CHANNEL) && defined(TENSOR_WIDTH) && defined(TENSOR_HEIGHT) */
185 
186  *offset_bottom = *offset_top + input->stride_y / sizeof(DATA_TYPE) - pad_horiz;
187 
188  return;
189 }

193 /** Performs a MAX pooling of pool size equal to 2, and record max value indices for NCHW.

196  * @note Pool sizes must be passed using -DPOOL_SIZE_X and -DPOOL_SIZE_Y e.g. -DPOOL_SIZE_X=13;
197  * @note Tensors width and height must be passed at compile time using -DMAX_WIDTH and -DMAX_HEIGHT
198  * @note Pool strides must be passed at compile time using -DSTRIDE_X and -DSTRIDE_Y which are the steps of the window along the x and y directions

200  *
201  * @param[in] input_ptr Pointer to the source tensor. Supported data types: F32
202  * @param[in] input_stride_x Stride of the source tensor in X dimension (in bytes)
203  * @param[in] input_step_x input_stride_x * number of elements along X processed per workitem(in bytes)
204  * @param[in] input_stride_y Stride of the source tensor in Y dimension (in bytes)
205  * @param[in] input_step_y input_stride_y * number of elements along Y processed per workitem(in bytes)
206  * @param[in] input_stride_z Stride of the source tensor in Z dimension (in bytes)
207  * @param[in] input_step_z input_stride_z * number of elements along Z processed per workitem(in bytes)
208  * @param[in] input_offset_first_element_in_bytes The offset of the first element in the source tensor
209  * @param[out] output_ptr Pointer to the destination tensor. Supported data types: same as @p input_ptr
210  * @param[in] output_stride_x Stride of the destination tensor in X dimension (in bytes)
211  * @param[in] output_step_x output_stride_x * number of elements along X processed per workitem(in bytes)
212  * @param[in] output_stride_y Stride of the destination tensor in Y dimension (in bytes)
213  * @param[in] output_step_y output_stride_y * number of elements along Y processed per workitem(in bytes)
214  * @param[in] output_stride_z Stride of the source tensor in Z dimension (in bytes)
215  * @param[in] output_step_z output_stride_z * number of elements along Z processed per workitem(in bytes)
216  * @param[in] output_offset_first_element_in_bytes The offset of the first element in the destination tensor
217  * @param[in] indices_ptr Pointer to the indices tensor. Supported data types: U32
218  * @param[in] indices_stride_x Stride of the indices tensor in X dimension (in bytes)
219  * @param[in] indices_step_x indices_stride_x * number of elements along X processed per workitem(in bytes)
220  * @param[in] indices_stride_y Stride of the indices tensor in Y dimension (in bytes)
221  * @param[in] indices_step_y indices_stride_y * number of elements along Y processed per workitem(in bytes)
222  * @param[in] indices_stride_z Stride of the indices tensor in Z dimension (in bytes)
223  * @param[in] indices_step_z indices_stride_z * number of elements along Z processed per workitem(in bytes)
224  * @param[in] indices_offset_first_element_in_bytes The offset of the first element in the indices tensor
225  */
226 __kernel void pooling_layer_2_nchw_indices_fp32(
227  TENSOR3D_DECLARATION(input),
228  TENSOR3D_DECLARATION(output),
229  TENSOR3D_DECLARATION(indices))
230 {
231  // Get pixels pointer
232  Tensor3D input = CONVERT_TO_TENSOR3D_STRUCT(input);
233  Tensor3D output = CONVERT_TO_TENSOR3D_STRUCT(output);
234  Tensor3D indices = CONVERT_TO_TENSOR3D_STRUCT(indices);
235 
236  // Load data
237  float2 data0 = VLOAD(2)(0, (__global float *)tensor3D_offset(&input, 0, 0, 0));
238  float2 data1 = VLOAD(2)(0, (__global float *)tensor3D_offset(&input, 0, 1, 0));
239 
240  // Perform calculations
241  float data0_max = POOL_OP(data0.s0, data0.s1);
242  float data1_max = POOL_OP(data1.s0, data1.s1);
243  float res = POOL_OP(data0_max, data1_max);
244  // Store result
245  *(__global float *)output.ptr = res;
246 
247 #if defined(PAD_TENSOR_LEFT) && defined(PAD_TENSOR_RIGHT) && defined(PAD_TENSOR_TOP) && defined(PAD_TENSOR_BOTTOM)
248 
249  uint offset_top = 0;
250  uint offset_bottom = 0;
251 
252  offset_no_padding_nchw(&input, &offset_top, &offset_bottom);
253 
254  uint index0 = select(offset_top + 1, offset_top, isgreaterequal(data0.s0, data0.s1));
255  uint index1 = select(offset_bottom + 1, offset_bottom, isgreaterequal(data1.s0, data1.s1));
256  uint index = select(index1, index0, isgreaterequal(data0_max, data1_max));
257 
258  *(__global uint *)indices.ptr = index;
259 
260 #endif //defined(PAD_TENSOR_LEFT) && defined(PAD_TENSOR_RIGHT) && defined(PAD_TENSOR_TOP) && defined(PAD_TENSOR_BOTTOM)
261 }
